using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Algorithms
{
    class Program
    {
        static void Main(string[] args)
        {
            string FullString = "Shehrozisagoodboy";
            string substr = "ali";
            if (IsContainSubstring(FullString, substr))
                Console.WriteLine("substr is present");
            else
                Console.WriteLine("substring is not present");


            Console.ReadKey();

        }
        public static bool IsContainSubstring(string FullString, string substring)
        {
            bool IsFound = false;
            for (int i = 0; i < FullString.Length; i++)
            {
                for (int j = 0; j < substring.Length; j++)
                {
                    if (FullString[i + j] == substring[j])
                    {
                        IsFound = true;
                    }
                    else
                    {
                        IsFound = false;
                        break;
                    }
                }
                if (IsFound)
                    return IsFound;
            }
            return IsFound;
        }
    }
}
